##### 弹幕效果，走马灯效果

https://github.com/chenjianfang/scroxt

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title></title>
    <style>
        .my-ele{
            width: 100px;
            height: 100px;
            margin-top:50px;
            margin-left: 50px;
            font-size: 16px;
            border: 1px solid;
            overflow: hidden;
        }
        .scroxt-vertical{
            border: 1px solid;
        }
        .btn{
            display: inline-block;
            width: 50px;
            height: 30px;
            line-height: 30px;
            text-align: center;
            border: 1px solid;
            cursor: pointer;
            margin-left: 10px;
            margin-top: 10px;
        }
    </style>
</head>
<body>

<!-- 滚动 -->
<div class="my-ele">
</div>
<div>
    <span class="start btn">开始</span>
    <span class="stop btn">暂停</span>
</div>
<script src="./js/scroxt.min.js"></script>
<script>
    var arr = [];
    for(var i = 0; i < 70; i++){
        arr.push('<div><p class="test">第'+i+'条</p><div>你好啊<div>你好啊<div>你好啊</div></div></div></div>');
    }
  //Horizontal横向水平，Vertical纵向垂直
    var scroxtVertical = new scroxt.Horizontal({
        target: ".my-ele",
        data: arr,
        speed: -5, // 滚动快慢
        gap: 20  // 每条数据间隔宽度，单位px                                  
    });
    //暂停
    document.querySelector(".stop").addEventListener("click",function(){
        scroxtVertical.stopMove();
    });
    //开始
    document.querySelector(".start").addEventListener("click",function(){
        scroxtVertical.startMove();
    });
    
</script>
</body>
</html>
```
